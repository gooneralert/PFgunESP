local RunService = game:GetService("RunService")
local Camera = workspace.CurrentCamera
local ESP_Utility = {}
ESP_Utility.__index = ESP_Utility

ESP_Utility.TrackersToUpdate = {}

local function WorldToScreen(position)
    local screenPos, onScreen = Camera:WorldToViewportPoint(position)
    return Vector2.new(screenPos.X, screenPos.Y), onScreen
end

local BasePartTypes = {
    ["Part"] = "BasePart", ["MeshPart"] = "BasePart", ["UnionOperation"] = "BasePart", ["Model"] = "Model",
}

local function IsValidObject(Object)
    if type(Object) == "userdata" and Object and Object.ClassName then 
        return BasePartTypes[Object.ClassName]
    end
    return nil
end

function ESP_Utility.NewTracker(Object, CustomName, Color)
    local ObjectType = IsValidObject(Object)
    if not ObjectType then return end 

    if ESP_Utility.TrackersToUpdate[Object] then return ESP_Utility.TrackersToUpdate[Object] end

    local self = setmetatable({}, ESP_Utility)
    self.Name = CustomName or Object.Name
    self.Object = Object
    self.Color = Color or Color3.fromRGB(255,255,255)
    self.Drawings = {}
    self.Visible = true

    -- Box
    self.Drawings["Box"] = Drawing.new("Square")
    self.Drawings["Box"].Thickness = 1
    self.Drawings["Box"].Filled = false
    self.Drawings["Box"].Color = self.Color

    -- Name
    self.Drawings["Name"] = Drawing.new("Text")
    self.Drawings["Name"].Size = 18
    self.Drawings["Name"].Center = true
    self.Drawings["Name"].Outline = true
    self.Drawings["Name"].Color = Color3.new(1,1,1)
    self.Drawings["Name"].Text = self.Name

    ESP_Utility.TrackersToUpdate[Object] = self
    return self
end

function ESP_Utility:_Update()
    if not self.Object or not self.Object.Parent then
        for _, d in pairs(self.Drawings) do d:Remove() end
        ESP_Utility.TrackersToUpdate[self.Object] = nil
        return 
    end

    local pos = self.Object.Position
    local size = self.Object.Size
    local screenPos, onScreen = WorldToScreen(pos)

    if onScreen then
        local top, topVisible = WorldToScreen(pos + Vector3.new(0, size.Y/2, 0))
        local bottom, botVisible = WorldToScreen(pos - Vector3.new(0, size.Y/2, 0))
        local height = math.abs(top.Y - bottom.Y) * 2
        local width = height * 1.2

        -- Update Box
        self.Drawings["Box"].Visible = true
        self.Drawings["Box"].Size = Vector2.new(width, height)
        self.Drawings["Box"].Position = Vector2.new(screenPos.X - width/2, screenPos.Y - height/2)

        -- Update Name (Check _G.name toggle)
        if _G.name == true then
            self.Drawings["Name"].Visible = true
            self.Drawings["Name"].Position = Vector2.new(screenPos.X, (screenPos.Y - height/2) - 20)
        else
            self.Drawings["Name"].Visible = false
        end
    else
        self.Drawings["Box"].Visible = false
        self.Drawings["Name"].Visible = false
    end
end

-- Render loop
RunService.RenderStepped:Connect(function()
    for _, tracker in pairs(ESP_Utility.TrackersToUpdate) do
        tracker:_Update()
    end
end)

-- Auto-track players loop
task.spawn(function()
    while task.wait(1) do
        for _, p in ipairs(game.Players:GetPlayers()) do
            if p ~= game.Players.LocalPlayer and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                ESP_Utility.NewTracker(p.Character.HumanoidRootPart, p.Name, Color3.fromRGB(255, 255, 255))
            end
        end
    end
end)

return ESP_Utility
