local RunService = game:GetService("RunService")
local Camera = workspace.CurrentCamera
local ESP_Utility = {}
ESP_Utility.__index = ESP_Utility

ESP_Utility.TrackersToUpdate = {}

local function WorldToScreen(position)
    local screenPos, onScreen = Camera:WorldToViewportPoint(position)
    return Vector2.new(screenPos.X, screenPos.Y), onScreen
end

function ESP_Utility.NewTracker(Object, CustomName, Color)
    -- Using .Address is critical for External LuaVMs to track memory pointers
    local addr = Object.Address 
    if not addr then return end
    if ESP_Utility.TrackersToUpdate[addr] then return ESP_Utility.TrackersToUpdate[addr] end

    local self = setmetatable({}, ESP_Utility)
    self.Name = CustomName or "Player"
    self.Object = Object
    self.Color = Color or Color3.fromRGB(255, 255, 255)
    self.Drawings = {}

    -- Initialize Box
    local box = Drawing.new("Square")
    box.Thickness = 1
    box.Filled = false
    box.Color = self.Color
    box.Visible = false
    self.Drawings["Box"] = box

    -- Initialize Name
    local name = Drawing.new("Text")
    name.Size = 18
    name.Center = true
    name.Outline = true
    name.Color = Color3.new(1, 1, 1)
    name.Visible = false
    self.Drawings["Name"] = name

    ESP_Utility.TrackersToUpdate[addr] = self
    return self
end

function ESP_Utility:_Update()
    -- Check if the object still exists in memory
    if not self.Object or not self.Object.Parent then
        for _, d in pairs(self.Drawings) do d.Visible = false d:Remove() end
        ESP_Utility.TrackersToUpdate[self.Object.Address] = nil
        return 
    end

    local pos = self.Object.Position
    local screenPos, onScreen = WorldToScreen(pos)

    if onScreen then
        -- Calculate size based on distance
        local camPos = Camera.CFrame.Position
        local dist = (camPos - pos).Magnitude
        local sizeMult = 2000 / dist
        local width, height = sizeMult, sizeMult * 1.5

        -- Update Box
        self.Drawings["Box"].Size = Vector2.new(width, height)
        self.Drawings["Box"].Position = Vector2.new(screenPos.X - width/2, screenPos.Y - height/2)
        self.Drawings["Box"].Visible = true

        -- Update Name Toggle
        if _G.name == true then
            self.Drawings["Name"].Text = self.Name
            self.Drawings["Name"].Position = Vector2.new(screenPos.X, (screenPos.Y - height/2) - 20)
            self.Drawings["Name"].Visible = true
        else
            self.Drawings["Name"].Visible = false
        end
    else
        self.Drawings["Box"].Visible = false
        self.Drawings["Name"].Visible = false
    end
end

-- Render loop to keep boxes stuck to players
RunService.RenderStepped:Connect(function()
    for _, tracker in pairs(ESP_Utility.TrackersToUpdate) do
        tracker:_Update()
    end
end)

-- Tracking Loop: Optimized for Externals
task.spawn(function()
    print("ESP Utility: Searching for targets...")
    while task.wait(2) do
        for _, p in ipairs(game.Players:GetPlayers()) do
            if p ~= game.Players.LocalPlayer then
                -- Check for standard character or PF's specific model setup
                local char = p.Character
                local root = char and (char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("RootPart") or char.PrimaryPart)
                
                if root then
                    ESP_Utility.NewTracker(root, p.Name, Color3.fromRGB(255, 255, 255))
                end
            end
        end
    end
end)

return ESP_Utility
